# At minimum, we need the tweet ID, retweet count and favorite count. I don't see any additional fields that could help us with the analysis, therefore not storing them into the dataframe
tweet_id = []
retweet_count = []
favorite_count = []
with open('tweet-json.txt') as f:
    line = f.readline()
    while(line):
        tweet = json.loads(line)
        tweet_id.append(tweet['id'])
        retweet_count.append(tweet['retweet_count'])
        favorite_count.append(tweet['favorite_count'])
        line = f.readline()

tweets_json_df = pd.DataFrame({'tweet_id':tweet_id,'retweet_count':retweet_count,'favorite_count':favorite_count})
